const axios = require("axios");

function extractId(url) {
  return url.split("/").pop();
}

// Archidekt
async function getArchidekt(url) {
  const id = extractId(url);
  const res = await axios.get(`https://archidekt.com/api/decks/${id}/`);
  return res.data.cards.map(c => c.card.oracleCard.name);
}

// Moxfield
async function getMoxfield(url) {
  const id = extractId(url);
  const res = await axios.get(`https://api.moxfield.com/v2/decks/all/${id}`);
  return Object.values(res.data.mainboard).map(c => c.card.name);
}

// EDHREC (basic scrape)
async function getEDHREC(url) {
  const commander = url.split("/").pop();
  const res = await axios.get(`https://edhrec.com/commanders/${commander}`);
  const matches = [...res.data.matchAll(/data-card-name="([^"]+)"/g)];
  return matches.map(m => m[1]);
}

// Text input
function parseText(text) {
  return text.split("\n")
    .map(l => l.replace(/^\d+\s*/, "").trim())
    .filter(Boolean);
}

async function parseInput(input) {
  if (input.includes("archidekt.com")) return await getArchidekt(input);
  if (input.includes("moxfield.com")) return await getMoxfield(input);
  if (input.includes("edhrec.com")) return await getEDHREC(input);
  return parseText(input);
}

module.exports = { parseInput };
